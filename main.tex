\frame{
\frametitle{Pros} 
\begin{itemize}
  \item Bigger hole for mounted small robots or unmounted big robot;
  \item Dry operation;
  \item Free access;
  \item \textbf{Already used for turbine maintenance}.
\end{itemize}
}

\frame{
\frametitle{Cons} 
\begin{itemize}
  \item Not big enough for mounted big robots;
  \item Complex transport: scaffolding and hoist;
  \item Difficulties for robot positioning and movement inside the
  Aro Camara. Slippery and sloping floor. 
\end{itemize}
}
